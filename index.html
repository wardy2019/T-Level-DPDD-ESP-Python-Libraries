Complete reference for ESP coding, debugging, and data analysis tasks.

Importing Data


import pandas as pd                # import pandas for data analysis
import matplotlib.pyplot as plt   # import matplotlib for graphs

df = pd.read_csv("data.csv")     # load CSV into DataFrame
print(df.head())                 # preview first 5 rows
print(df.info())                 # view structure and missing values
print(df.describe())             # view statistics

Selecting and Filtering


df["Column"]                     # select a single column

df[["Column1", "Column2"]]       # select multiple columns

df[df["Column"] == "Value"]      # filter rows by value

df[(df["Column"] == "A") & (df["Other"] == "B")]   # multiple conditions

df.loc[:, ["Column1", "Column2"]]   # select specific columns safely

Counting and Sorting


df["Item"].value_counts()        # count occurrences

df["Item"].unique()              # show unique values

df.sort_values("Value", ascending=False)   # sort highest to lowest

df["Value"].idxmax()             # index of highest value

df.loc[df["Value"].idxmax()]     # show row with highest value

Creating Columns


df["NewColumn"] = df["Value"] * 1.2   # create calculated column

df["IsLarge"] = df["Value"] > 50      # create boolean column

Data Cleaning


df.isnull().sum()               # check missing values

df.fillna(0, inplace=True)      # replace missing values

df.dropna(inplace=True)         # remove missing rows

df.drop(columns=["Column"], inplace=True)   # delete column

df.rename(columns={"Old": "New"}, inplace=True)   # rename column

subset = df[df["Category"] == "A"].copy()   # create safe subset before editing

Grouping and Aggregation


df.groupby("Category")["Value"].sum()   # total per category

df.groupby("Category")["Value"].mean()  # average per category

df.groupby("Category")["Value"].count() # count per category

df.groupby("Category")["Value"].max()   # highest per category

df.groupby("Category")["Value"].min()   # lowest per category

result = df.groupby("Category")["Value"].sum().reset_index()
# turn grouped data back into table

df.groupby("Category")["Value"].sum().sort_values(ascending=False)
# group then sort

df.sort_values(["Column1","Column2"], ascending=[True, False])
# sort by multiple columns

Working with Dates


df["Date"] = pd.to_datetime(df["Date"])   # convert to datetime

df["Weekday"] = df["Date"].dt.day_name()  # extract weekday name

trend = df.groupby("Date")["Value"].sum().sort_index()
# grouped data sorted to show trends over time

Error Handling (try/except)


try:
    df = pd.read_csv("data.csv")
except FileNotFoundError:
    print("File not found")

try:
    print(df["Sales"])
except KeyError:
    print("Column not found")

try:
    choice = int(input("Enter number: "))
except ValueError:
    print("Invalid input")

Graphing


df.groupby("Category")["Value"].sum().plot(kind="bar")
plt.title("Totals by Category")
plt.xlabel("Category")
plt.ylabel("Value")
plt.xticks(rotation=45)
plt.legend()
plt.show()

df.groupby("Date")["Value"].sum().plot(kind="line")
plt.title("Trend over time")
plt.grid(True)
plt.show()

df["Category"].value_counts().plot(kind="pie", autopct="%1.1f%%")
plt.title("Category Share")
plt.show()

df.loc[0, "Value"]              # access by label

df.iloc[0:3, 1:3]               # access by position
