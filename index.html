Getting Ready for Each Lesson

Steps before you start coding.

# Import libraries with short aliases
import pandas as pd
import matplotlib.pyplot as plt

1. Importing, Reading, and Understanding Data

# Read data from a CSV file into a DataFrame
df = pd.read_csv("your_data.csv")

# Show the first 5 rows
print(df.head())

# Show column info (types and counts)
print(df.info())

# Show summary statistics
print(df.describe())

2. Selecting, Filtering, and Creating Subsets

# Select a single column
df["Column"]

# Filter rows where column equals a value
df[df["Column"] == "Value"]

# Filter with multiple conditions
df[(df["Column"] == "X") & (df["Other"] == "Y")]

# Create a subset DataFrame
subset = df[df["Category"] == "X"]

3. Counting, Sorting, and Finding Extremes

# Count frequency of values
df["Item"].value_counts()

# List unique values
df["Item"].unique()

# Sort by a column
df.sort_values("Value", ascending=False)

# Index of highest value
df["Value"].idxmax()

# Row at that index
df.loc[df["Value"].idxmax()]

4. Creating New Columns

# Create a calculated column
df["NewColumn"] = df["Value"] * 1.2

# Create a True/False column
df["IsLarge"] = df["Value"] > 50

5. Grouping and Summarising Data

# Group by one column and sum
df.groupby("Category")["Value"].sum()

# Group by two columns and calculate average
df.groupby(["Date", "Category"])["Value"].mean()

6. Working with Dates

# Convert column to datetime
df["Date"] = pd.to_datetime(df["Date"])

# Create weekday column
df["Weekday"] = df["Date"].dt.day_name()

# Convert input into datetime
end_date = input("Enter a date: ")
pd.to_datetime(end_date, dayfirst=True)

7. Input Validation and Error Handling

try:
    # Convert input to integer
    user_input = int(input("Enter a number: "))
except ValueError:
    # Handle invalid input
    print("Invalid input")

8. Menus and Mapping Choices

# Print a simple menu
print("1. Option A")
print("2. Option B")

# Store options in a list
options = ["Option A", "Option B"]

# Get user choice
choice = int(input("Choose 1 or 2: "))

# Map choice to option
selected = options[choice - 1]

9. Using Functions and Adding Comments

# Define a reusable function
def show_totals():
    # Print total of 'Value' column
    print(df["Value"].sum())

# Call the function
show_totals()

10. Formatting Output for the User

total = 250

# Using format()
print("Total: £{:.2f}".format(total))

# Using f-string
print(f"Total: £{total:.2f}")

11. Graphing with Matplotlib

# Bar chart
df.groupby("Category")["Value"].sum().plot(kind="bar")
plt.title("Total by Category")
plt.xlabel("Category")
plt.ylabel("Value")
plt.tight_layout()
plt.show()

# Line chart
df.groupby("Date")["Value"].sum().plot(kind="line")
plt.title("Value Over Time")
plt.xlabel("Date")
plt.ylabel("Value")
plt.grid(True)
plt.tight_layout()
plt.show()

# Pie chart
df["Category"].value_counts().plot(kind="pie", autopct="%1.1f%%")
plt.title("Category Share")
plt.tight_layout()
plt.show()

12. loc and iloc

# Access row and column by label
df.loc[5, "Value"]

# Access by index positions
df.iloc[0:3, 1:3]

13. Statistical Methods

# Calculate statistics
mean_value = df["Value"].mean()
min_value = df["Value"].min()
max_value = df["Value"].max()
sum_value = df["Value"].sum()

14. idxmax and idxmin

# Index of maximum
max_index = df["Value"].idxmax()

# Index of minimum
min_index = df["Value"].idxmin()

15. Groupby with Sort

# Group and calculate mean
average = df.groupby(["Menu Item", "Service"]).mean()

# Get unique months and sort them
months = df["Month"].unique()
months.sort()
