Error Handling (try/except)


# File errors
try:
    df = pd.read_csv("data.csv")
except FileNotFoundError:
    print("File not found")

# Column errors
try:
    print(df["Sales"])
except KeyError:
    print("Column not found")

# Input errors
try:
    choice = int(input("Enter number: "))
except ValueError:
    print("Invalid input")
